## 1. I18n管理器

```java
public class I18nManager {
    private static I18nManager instance;
    private ResourceBundle bundle;
    private Locale currentLocale;
    private final Map<Locale, ResourceBundle> bundles = new HashMap<>();
    private final List<LocaleChangeListener> listeners = new ArrayList<>();
    
    private I18nManager() {
        loadBundle(Locale.CHINESE);
    }
    
    public static I18nManager getInstance() {
        if (instance == null) {
            instance = new I18nManager();
        }
        return instance;
    }
    
    public void setLocale(Locale locale) {
        if (!locale.equals(currentLocale)) {
            loadBundle(locale);
            notifyListeners();
        }
    }
    
    private void loadBundle(Locale locale) {
        if (bundles.containsKey(locale)) {
            bundle = bundles.get(locale);
        } else {
            bundle = ResourceBundle.getBundle("i18n.messages", locale);
            bundles.put(locale, bundle);
        }
        currentLocale = locale;
    }
    
    public String get(String key) {
        try {
            return bundle.getString(key);
        } catch (MissingResourceException e) {
            return key;
        }
    }
    
    public String get(String key, Object... args) {
        String template = get(key);
        return MessageFormat.format(template, args);
    }
    
    public ResourceBundle getBundle() {
        return bundle;
    }
    
    public Locale getCurrentLocale() {
        return currentLocale;
    }
    
    public List<Locale> getSupportedLocales() {
        return List.of(
            Locale.CHINESE,
            Locale.US
        );
    }
    
    public void addListener(LocaleChangeListener listener) {
        listeners.add(listener);
    }
    
    private void notifyListeners() {
        for (LocaleChangeListener listener : listeners) {
            listener.onLocaleChanged(currentLocale);
        }
    }
    
    public interface LocaleChangeListener {
        void onLocaleChanged(Locale newLocale);
    }
}
```

## 3. 动态语言切换

```java
public class LocaleChangeListener implements I18nManager.LocaleChangeListener {
    private final Node node;
    private final String key;
    
    public LocaleChangeListener(Node node, String key) {
        this.node = node;
        this.key = key;
    }
    
    @Override
    public void onLocaleChanged(Locale newLocale) {
        Platform.runLater(() -> {
            if (node instanceof Label label) {
                label.setText(I18nManager.getInstance().get(key));
            } else if (node instanceof Button button) {
                button.setText(I18nManager.getInstance().get(key));
            }
            // ... 其他节点类型
        });
    }
}

public class I18nBinder {
    public static void bind(Label label, String key) {
        label.setText(I18nManager.getInstance().get(key));
        I18nManager.getInstance().addListener(new LocaleChangeListener(label, key));
    }
    
    public static void bind(Button button, String key) {
        button.setText(I18nManager.getInstance().get(key));
        I18nManager.getInstance().addListener(new LocaleChangeListener(button, key));
    }
    
    public static void bindTitle(Stage stage, String key) {
        stage.titleProperty().bind(Bindings.createStringBinding(
            () -> I18nManager.getInstance().get(key),
            I18nManager.getInstance().localeProperty()
        ));
    }
}
```

## 6. 语言选择器组件

```java
public class LanguageSelector extends HBox {
    @FXML private ComboBox<Locale> languageBox;
    
    public LanguageSelector() {
        loadFxml();
        initialize();
    }
    
    private void initialize() {
        languageBox.getItems().addAll(I18nManager.getInstance().getSupportedLocales());
        languageBox.setCellFactory(list -> new LocaleListCell());
        languageBox.setButtonCell(new LocaleListCell());
        languageBox.setValue(I18nManager.getInstance().getCurrentLocale());
        
        languageBox.getSelectionModel().selectedItemProperty().addListener((obs, old, newVal) -> {
            I18nManager.getInstance().setLocale(newVal);
        });
    }
    
    private static class LocaleListCell extends ListCell<Locale> {
        @Override
        protected void updateItem(Locale locale, boolean empty) {
            super.updateItem(locale, empty);
            if (empty || locale == null) {
                setText(null);
                setGraphic(null);
            } else {
                setText(locale.getDisplayName(locale));
                // 可以添加国旗图标
            }
        }
    }
}
```
