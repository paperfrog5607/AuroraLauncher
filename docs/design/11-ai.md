# 模块十一：ai - AI服务模块

## 1. 模块概述

多AI提供者支持，聊天、翻译、崩溃分析、脚本生成、模组推荐、代码审查。

public interface AiService {
    String getName();
    CompletableFuture<AiResponse> chat(List<ChatMessage> messages, AiOptions options);
    CompletableFuture<AiResponse> complete(String prompt, AiOptions options);
    boolean isAvailable();
    int getMaxTokens();
}

public abstract class AiProvider implements AiService {
    protected final AiConfig config;
    protected final HttpClient httpClient;
    
    protected abstract HttpRequest buildRequest(List<ChatMessage> messages, AiOptions options);
    protected abstract AiResponse parseResponse(String responseBody);
    
    @Override
    public CompletableFuture<AiResponse> chat(List<ChatMessage> messages, AiOptions options) {
        HttpRequest request = buildRequest(messages, options);
        return httpClient.sendAsync(request, BodyHandlers.ofString())
            .thenApply(this::parseResponse);
    }
}

    private String apiKey;
    private String baseUrl;

}

public class OpenAIProvider extends AiProvider {
    private static final String API_URL = "https://api.openai.com/v1/chat/completions";
    
    public OpenAIProvider(AiConfig config) {
        super(config);
    }
    
    @Override
    protected HttpRequest buildRequest(List<ChatMessage> messages, AiOptions options) {
        JsonObject body = new JsonObject();
        body.addProperty("model", config.getModel());
        body.add("messages", buildMessagesArray(messages));
        if (options.getTemperature() != null) {
            body.addProperty("temperature", options.getTemperature());
        }
        if (options.getMaxTokens() != null) {
            body.addProperty("max_tokens", options.getMaxTokens());
        }
        
        return HttpRequest.newBuilder()
            .uri(URI.create(API_URL))
            .header("Authorization", "Bearer " + config.getApiKey())
            .header("Content-Type", "application/json")
            .POST(HttpRequest.BodyPublishers.ofString(body.toString()))
            .build();
    }
}

### 4.3 ChatService

```java
public class ChatService {
    private final AiProvider provider;
    private final Map<String, ChatSession> sessions;
    private final SystemPromptRegistry promptRegistry;
    
    public CompletableFuture<AiResponse> sendMessage(String sessionId, String message) {
        ChatSession session = sessions.computeIfAbsent(sessionId, ChatSession::new);
        session.addMessage(ChatMessage.user(message));
        
        List<ChatMessage> messages = buildMessages(session);
        return provider.chat(messages, new AiOptions())
            .thenApply(response -> {
                session.addMessage(ChatMessage.assistant(response.getContent()));
                return response;
            });
    }
    
    public void clearSession(String sessionId) {
        sessions.remove(sessionId);
    }
    
    public ChatHistory getHistory(String sessionId) {
        return sessions.get(sessionId).getHistory();
    }
    
    public void setSystemPrompt(String sessionId, String promptName) {
        SystemPrompt prompt = promptRegistry.get(promptName);
        sessions.get(sessionId).setSystemPrompt(prompt);
    }
}

public class ChatSession {
    private String id;
    private SystemPrompt systemPrompt;
    private ChatHistory history;
    private Instant createdTime;
    private Instant lastActiveTime;
    
    public void addMessage(ChatMessage message) {
        history.add(message);
        lastActiveTime = Instant.now();
    }
}

public class ChatHistory {
    private List<ChatMessage> messages;
    private int maxMessages = 100;
    
    public void add(ChatMessage message) {
        messages.add(message);
        if (messages.size() > maxMessages) {
            messages.remove(0);
        }
    }
    
    public List<ChatMessage> getRecent(int count) {
        return messages.subList(Math.max(0, messages.size() - count), messages.size());
    }
}

public class SystemPrompt {
    private String name;
    private String template;
    private Map<String, String> variables;
    
    public String render() {
        String result = template;
        for (Map.Entry<String, String> entry : variables.entrySet()) {
            result = result.replace("{{" + entry.getKey() + "}}", entry.getValue());
        }
        return result;
    }
}

### 4.7 ModRecommender

```java
public class ModRecommender {
    private final AiProvider provider;
    private final ModrinthClient modrinth;
    
    public CompletableFuture<List<RecommendationResult>> recommend(
            RecommendationContext context) {
        String prompt = buildPrompt(context);
        return provider.complete(prompt, RecommendationOptions())
            .thenApply(this::parseRecommendations)
            .thenCompose(this::enrichWithModData);
    }
    
    private String buildPrompt(RecommendationContext context) {
        StringBuilder sb = new StringBuilder();
        sb.append("Based on the following preferences, recommend 10 Minecraft mods:\n");
        sb.append("Play style: ").append(context.getPlayStyle()).append("\n");
        sb.append("Minecraft version: ").append(context.getMcVersion()).append("\n");
        sb.append("Mod loader: ").append(context.getLoader()).append("\n");
        if (!context.getExistingMods().isEmpty()) {
            sb.append("Already installed: ").append(formatMods(context.getExistingMods())).append("\n");
        }
        sb.append("Preferences: ").append(String.join(", ", context.getPreferences())).append("\n");
        return sb.toString();
    }
    
    private CompletableFuture<List<RecommendationResult>> enrichWithModData(
            List<RecommendationResult> recommendations) {
        // 从Modrinth获取详细信息
        List<CompletableFuture<RecommendationResult>> futures = recommendations.stream()
            .map(r -> modrinth.search(r.getName())
                .thenApply(results -> results.isEmpty() ? r : enrich(r, results.get(0))))
            .toList();
        
        return CompletableFuture.allOf(futures.toArray(new CompletableFuture[0]))
            .thenApply(v -> futures.stream().map(CompletableFuture::join).toList());
    }
}

public class RecommendationContext {
    private String playStyle;  // tech, magic, adventure, building
    private String mcVersion;
    private String loader;
    private List<String> existingMods;
    private List<String> preferences;
    private List<String> avoidCategories;
}

public class RecommendationResult {
    private String name;
    private String modId;
    private String description;
    private String reason;
    private String source;  // "modrinth", "curseforge"
    private String url;
    private String iconUrl;
}

## 5. 关键流程

### 5.1 聊天服务流程

```
用户发送消息
    ↓
获取或创建会话
    ↓
添加到历史记录
    ↓
构建请求:
├── 系统提示词
├── 历史消息
└── 当前消息
    ↓
调用AI API
    ↓
解析响应
    ↓
添加到历史记录
    ↓
返回响应
```

### 5.2 崩溃分析流程

```
接收崩溃日志
    ↓
构建分析提示词
    ↓
调用AI API
    ↓
解析结构化结果:
├── 崩溃类型
├── 根本原因
├── 可疑模组
└── 解决方案
    ↓
返回分析结果
```

### 5.3 模组推荐流程

```
收集用户偏好
    ↓
构建推荐提示词
    ↓
调用AI API获取推荐列表
    ↓
解析模组名称
    ↓
查询Modrinth/CurseForge:
├── 获取项目ID
├── 获取描述
├── 获取图标
└── 获取下载链接
    ↓
返回丰富后的推荐结果
```
